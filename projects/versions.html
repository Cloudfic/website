terraform {
  required_version = ">= 1.9.8"

  required_providers {
    google = {
      source  = "hashicorp/google"
      version = ">= 5.0.0, < 7.0.0"
    }
  }
}

provider "google" {
  project     = var.project_id         # The project_id variable should be defined in a separate variables.tf file or passed in via the command line.
  region      = var.region             # var.region should be defined in a variables.tf file or passed as a command-line variable. Expected value: the region where the resources will be deployed.
  credentials = var.GOOGLE_CREDENTIALS # var.GOOGLE_CREDENTIALS should be defined in a variables.tf file or passed as a command-line variable.
}
